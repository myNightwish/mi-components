# 快速开始

# 安装

npm install mi-components

# 引用组件库

在 main.js 中引用组件库

// 全部引用
import MiComponents from 'mi-components'
import 'mi-components/dist/css/index.css'
Vue.use(MiComponents)

// 按需引用
import { miCard } from 'mi-components'
import 'mi-components/dist/css/button.css'
Vue.use(miCard)
